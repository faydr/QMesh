106  static bool getBit(const vector<uint8_t> &bytes, const size_t pos) {
107  uint8_t byte = bytes[pos/8];
108  size_t byte_pos = pos % 8;
109  return ((((1 << byte_pos) & byte) == 0) ? false : true);
110  }
111 
112  static void setBit(const bool bit, const size_t pos, vector<uint8_t> &bytes) {
113  size_t byte_pos = pos % 8;
114  bytes[pos/8] &= ~(1 << byte_pos);
115  uint8_t my_bit = (bit == false) ? 0 : 1;
116  bytes[pos/8] |= (my_bit << byte_pos);
117  }

89  static bool getBit(const vector<uint8_t> &bytes, const size_t pos) {
90  uint8_t byte = bytes[pos/8];
91  uint8_t byte_pos = pos % 8;
92  return ((1 << byte_pos) & byte) == 0 ? false : true;
93  }
94 
95  static void setBit(const bool bit, const size_t pos, vector<uint8_t> &bytes) {
96  bytes[pos/8] &= ~(1 << pos);
97  uint8_t my_bit = bit == false ? 0 : 1;
98  bytes[pos/8] |= (my_bit << pos);
99  }
100 
101  static void interleaveBits(const vector<uint8_t> &bytes, vector<uint8_t> &int_bytes) {
102  int_bytes.resize(bytes.size());
103  size_t non_int_pos = 0;
104  for(size_t cur_bit = 0; cur_bit < 8; cur_bit++) {
105  for(size_t cur_byte = 0; cur_byte < bytes.size(); cur_byte++) {
106  bool bit = getBit(bytes, non_int_pos);
107  setBit(bit, cur_byte*8+cur_bit, int_bytes);
108  non_int_pos += 1;
109  }
110  }
111  }
112 
113  static void deinterleaveBits(const vector<uint8_t> &int_bytes, vector<uint8_t> &bytes) {
114  bytes.resize(int_bytes.size());
115  size_t non_int_pos = 0;
116  for(size_t cur_bit = 0; cur_bit < 8; cur_bit++) {
117  for(size_t cur_byte = 0; cur_byte < bytes.size(); cur_byte++) {
118  bool bit = getBit(int_bytes, cur_byte*8+cur_bit);
119  setBit(bit, non_int_pos, bytes);
120  non_int_pos += 1;
121  }
122  }
123  }

222  size_t encode(const vector<uint8_t> &msg, vector<uint8_t> &enc_msg) {
223  enc_msg.resize(FECConv::getEncSize(msg.size()));
224  vector<uint8_t> msg_int(msg.size());
225  interleaveBits(msg, msg_int);
226  return correct_convolutional_encode(corr_con, msg_int.data(), msg_int.size(), enc_msg.data())/8;
227  }
228 
229  ssize_t decode(const vector<uint8_t> &enc_msg, vector<uint8_t> &dec_msg) {
230  dec_msg.resize(enc_msg.size());
231  vector<uint8_t> dec_msg_int(dec_msg.size());
232  size_t dec_size = correct_convolutional_decode(corr_con, enc_msg.data(),
233  enc_msg.size()*8, dec_msg_int.data());
234  dec_msg.resize(dec_size);
235  deinterleaveBits(dec_msg_int, dec_msg);
236  return dec_size;
237  }

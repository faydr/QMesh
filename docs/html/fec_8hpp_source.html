138 
139  static bool getBit(const vector<uint8_t> &bytes, const int32_t pos) {
140  uint8_t byte = bytes[pos/8];
141  size_t byte_pos = pos % 8;
142  return ((((1 << byte_pos) & byte) == 0) ? false : true);
143  }
144 
145  static void setBit(const bool bit, const int32_t pos, vector<uint8_t> &bytes) {
146  size_t byte_pos = pos % 8;
147  bytes[pos/8] &= ~(1 << byte_pos);
148  uint8_t my_bit = (bit == false) ? 0 : 1;
149  bytes[pos/8] |= (my_bit << byte_pos);
150  }

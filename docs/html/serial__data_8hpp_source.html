408 template <class T>
409 void enqueue_mail(Mail<T, QUEUE_DEPTH> &mail_queue, T val) {
410  auto mail_item = mail_queue.alloc_for(osWaitForever);
411  MBED_ASSERT(mail_item != NULL);
412  *mail_item = val;
413  while(mail_queue.full());
414  mail_queue.put(mail_item);

440 template <class T>
441 T dequeue_mail(Mail<T, QUEUE_DEPTH> &mail_queue) {
442  T mail_item;
443  for(;;) {
444  osEvent evt = mail_queue.get();
445  if(evt.status == osEventMail) {
446  mail_item = *((T *) evt.value.p);
447  mail_queue.free((T *) evt.value.p);
448  break;
449  }
450  else { MBED_ASSERT(false); }
451  }
452  return mail_item;
453 }

455 template <class T>
456 T dequeue_mail_timeout(Mail<T, QUEUE_DEPTH> &mail_queue, const uint32_t timeout_ms, bool &timed_out) {
457  T mail_item;
458  timed_out = false;
459  osEvent evt = mail_queue.get(timeout_ms);
460  if(evt.status == osEventMail) {
461  mail_item = *((T *) evt.value.p);
462  mail_queue.free((T *) evt.value.p);
463  }
464  else if(evt.status == osEventTimeout) {
465  timed_out = true;
466  }
467  else { MBED_ASSERT(false); }
468  return mail_item;
469 }
